/* تشغيل الصوت */
const music = document.getElementById("music");
document.body.addEventListener("click", () => {
  music.play().catch(()=>{});
}, { once: true });

/* العداد */
const startDate = new Date("2023-12-14");

function updateCounter() {
  const diff = Date.now() - startDate.getTime();
  const sec = Math.floor(diff/1000);
  const d = Math.floor(sec/86400);
  const h = Math.floor((sec%86400)/3600);
  const m = Math.floor((sec%3600)/60);
  const s = sec%60;

  document.getElementById("counter").innerHTML =
  `${d} يوم - ${h} ساعة - ${m} دقيقة - ${s} ثانية`;
}

setInterval(updateCounter,1000);
updateCounter();

/* رسائل */
const msgs = [
"بحبك ❤️",
"أنتِ حياتي 💖",
"أجمل حاجة في حياتي 🌹",
"معاك الدنيا أحلى 💕",
"وجودك في حياتي نعمة ❤️"
];

function randomMsg(){
  document.getElementById("randomMessage").innerText =
  msgs[Math.floor(Math.random()*msgs.length)];
}
setInterval(randomMsg,3000);
randomMsg();

/* زر بحبك */
let love=0;
function addLove(){
  love++;
  document.getElementById("loveCounter").innerText = love+" ❤️";

  for(let i=0;i<10;i++){
    const h=document.createElement("div");
    h.className="heart";
    h.innerHTML="❤️";
    h.style.left=Math.random()*100+"vw";
    document.body.appendChild(h);
    setTimeout(()=>h.remove(),4000);
  }
}

/* موسيقى */
function toggleMusic(){
  if(music.paused) music.play();
  else music.pause();
}

  m.style.display = m.style.display=="block"?"none":"block";

/* ألعاب نارية */
function fireworks(){
  for(let i=0;i<50;i++){
    setTimeout(()=>{
      const f=document.createElement("div");
      f.innerHTML="✨";
      f.style.position="fixed";
      f.style.left=Math.random()*100+"vw";
      f.style.top=Math.random()*100+"vh";
      document.body.appendChild(f);
      setTimeout(()=>f.remove(),2000);
    },i*20);
  }
}
